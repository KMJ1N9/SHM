#### 2.3 前端 — HTTP 客户端封装

**文件：`miniprogram/api/http.js`**（新建）

```ts
// uni.request 包装器，提供：
// 1. 自动拼接 base URL
// 2. 自动附加 Authorization header
// 3. 401 → 自动调用 refresh → 重试原请求（互斥锁防并发刷新）
// 4. 统一错误处理（code !== 0 → reject）
// 5. 请求/响应拦截器

interface HttpConfig {
  url: string;
  method?: 'GET' | 'POST' | 'PUT' | 'DELETE';
  data?: Record<string, unknown>;
  header?: Record<string, string>;
  skipAuth?: boolean;
}

async function request<T = unknown>(config: HttpConfig): Promise<T>
```

**关键实现：**

```ts
// Token 刷新互斥锁
let refreshing = false;
let refreshQueue: Array<{ resolve: Function; reject: Function }> = [];

async function refreshToken(): Promise<string> {
  if (refreshing) {
    return new Promise((resolve, reject) => {
      refreshQueue.push({ resolve, reject });
    });
  }
  refreshing = true;
  try {
    const refreshToken = uni.getStorageSync('refresh_token');
    const res = await uni.request({
      url: `${BASE_URL}/auth/refresh`,
      method: 'POST',
      data: { refresh_token: refreshToken },
    });
    const { access_token } = res.data.data;
    uni.setStorageSync('access_token', access_token);
    refreshQueue.forEach(q => q.resolve(access_token));
    refreshQueue = [];
    return access_token;
  } catch (err) {
    refreshQueue.forEach(q => q.reject(err));
    refreshQueue = [];
    // 跳转登录页
    uni.reLaunch({ url: '/pages/auth/login' });
    throw err;
  } finally {
    refreshing = false;
  }
}
```

#### 2.5 前端 — Pinia User Store

**文件：`miniprogram/store/user.js`**（替换占位）

```ts
import { defineStore } from 'pinia';
import { login as loginApi, getMe } from '@/api/auth';

interface UserState {
  token: string | null;
  refreshToken: string | null;
  userInfo: UserInfo | null;
  isLogin: boolean;
}

export const useUserStore = defineStore('user', {
  state: (): UserState => ({
    token: uni.getStorageSync('access_token') || null,
    refreshToken: uni.getStorageSync('refresh_token') || null,
    userInfo: null,
    isLogin: false,
  }),

  getters: {
    userId: (state) => state.userInfo?.id,
    creditScore: (state) => state.userInfo?.credit_score ?? 100,
    canPublish: (state) => (state.userInfo?.credit_score ?? 100) >= 60,
  },

  actions: {
    async login(code: string) {
      const res = await loginApi(code);
      this.token = res.access_token;
      this.refreshToken = res.refresh_token;
      this.userInfo = res.user;
      this.isLogin = true;
      uni.setStorageSync('access_token', res.access_token);
      uni.setStorageSync('refresh_token', res.refresh_token);
    },

    async fetchUserInfo() {
      this.userInfo = await getMe();
      this.isLogin = true;
    },

    logout() {
      this.token = null;
      this.refreshToken = null;
      this.userInfo = null;
      this.isLogin = false;
      uni.removeStorageSync('access_token');
      uni.removeStorageSync('refresh_token');
    },
  },
});
```

#### 11.2 后端 — 缓存接入

在以下 service 方法中接入 LRU 缓存：

| 方法 | 缓存 Key | TTL | 原因 |
|------|---------|:--:|------|
| `productService.list` | `products:list:${hash(filters)}` | 60s | 首页高频访问 |
| `productService.detail` | `product:${id}` | 120s | 商品详情页高频 |
| `userService.getById` | `user:public:${id}` | 300s | 用户信息变化少 |

**缓存失效策略：**
- 商品创建/更新/删除 → 清除 `products:list:*` + `product:${id}`
- 用户资料更新 → 清除 `user:public:${id}`

#### 11.3 后端 — 游标分页（商品列表）

当前使用偏移分页（`LIMIT ? OFFSET ?`）。在第 4 轮搜索中已可用。本轮增加游标分页支持以应对数据量增长后的性能问题：

```js
// 偏移分页（保留，用于管理后台等需要跳页的场景）
GET /api/products?page=1&pageSize=20

// 游标分页（新增，用于"加载更多"场景）
GET /api/products?cursor=123&limit=20  // cursor = 上一页最后一条的 id
```

在 `repository/product.js` 中增加 `listByCursor` 方法，`services/product.js` 中根据 Query 参数自动选择分页策略。

## 附录 B：快速启动第一轮

```bash
# 1. 环境变量
cd server
cp .env.example .env.development
# 编辑 .env.development：填入 MySQL 密码 + 微信 AppSecret + COS/IM 密钥

# 2. 数据库
mysql -u root -p -e "CREATE DATABASE campus_market_dev CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci"
npm run db:migrate
npm run db:seed

# 3. 启动
npm run dev
# → http://localhost:3000/api/health → {"code":0,"message":"ok","data":{"status":"healthy"}}

# 4. 验证前端编译
cd ../miniprogram
npm run dev:mp-weixin
# → 微信开发者工具打开 dist/dev/mp-weixin/
```
